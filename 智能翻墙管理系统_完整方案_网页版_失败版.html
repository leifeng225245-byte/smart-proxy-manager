🎯 问题诊断

原始问题

用户反馈: 关闭mihomo翻墙服务后，Ubuntu无法直连上网

症状: 所有网络应用都无法访问互联网

影响: 用户体验差，无法灵活切换网络模式

根本原因分析

  1. mihomo服务停止 ✅
  2. 环境变量残留 ❌
  3. 应用尝试连接已关闭的代理端口 ❌
  4. 网络请求失败 ❌

#!/bin/bash # ============================================================================= # 智能翻墙管理脚本 # 功能: 开关mihomo服务 + 管理环境变量 + 自动备份 + 状态记忆 # 作者: Cascade AI Assistant # 版本: v1.0 # 时间: 2026-03-15 04:03 # ============================================================================= # 颜色定义 RED='\033[0;31m' GREEN='\033[0;32m' YELLOW='\033[1;33m' BLUE='\033[0;34m' NC='\033[0m' # 配置文件路径 MIHOMO_CONFIG="$HOME/.config/mihomo/config.yaml" BASHRC="$HOME/.bashrc" STATUS_FILE="$HOME/.mihomo_status" BACKUP_DIR="/media/lsk/future/proxy-config-backup/smart-backup" LOG_FILE="$BACKUP_DIR/logs/mihomo-manager.log" # 代理配置 PROXY_HOST="127.0.0.1" PROXY_PORT="7890" # 创建必要的目录 mkdir -p "$BACKUP_DIR"/{configs,logs,weekly,emergency} # 日志函数 log() { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE" } # 显示帮助信息 show_help() { echo -e "${BLUE}智能翻墙管理脚本${NC}" echo "用法: $0 [命令]" echo "" echo "命令:" echo " on 开启翻墙" echo " off 关闭翻墙" echo " status 查看状态" echo " backup 手动备份" echo " restore 恢复配置" echo " test 测试网络" echo " help 显示帮助" } # 开启翻墙 enable_mihomo() { echo -e "${BLUE}开启翻墙...${NC}" backup_config sudo systemctl start mihomo sleep 2 enable_proxy_env echo "enabled" > "$STATUS_FILE" test_network log "翻墙已开启" } # 关闭翻墙 disable_mihomo() { echo -e "${BLUE}关闭翻墙...${NC}" backup_config sudo systemctl stop mihomo disable_proxy_env echo "disabled" > "$STATUS_FILE" if ping -c 1 -W 3 www.baidu.com >/dev/null 2>&1; then echo -e "${GREEN}🎉 翻墙已关闭，直连网络正常！${NC}" fi log "翻墙已关闭" } # 主函数 main() { case "${1:-help}" in "on") enable_mihomo ;; "off") disable_mihomo ;; "status") show_status ;; "backup") backup_config ;; "test") test_network ;; "help"|*) show_help ;; esac } main "$@"
