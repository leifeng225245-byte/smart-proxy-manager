### 🔍 问题诊断

#### 原始问题
```
用户反馈: 关闭mihomo翻墙服务后，Ubuntu无法直连上网
症状: 所有网络应用都无法访问互联网
影响: 用户体验差，无法灵活切换网络模式
```

#### 根本原因分析
```
1. mihomo服务停止 ✅
2. 环境变量残留 ❌
3. 应用尝试连接已关闭的代理端口 ❌
4. 网络请求失败 ❌
```

```bash

# 问题流程:
用户登录 → 加载~/.bashrc → 设置环境变量 → 应用使用代理
关闭mihomo → 服务停止 → 环境变量仍存在 → 应用连接失败
```

## 📱 完整代码实现

### 🎯 主脚本: mihomo-manager.sh

```bash
#!/bin/bash

# =============================================================================
# 智能翻墙管理脚本
# 功能: 开关mihomo服务 + 管理环境变量 + 自动备份 + 状态记忆
# 作者: Cascade AI Assistant
# 版本: v1.0
# 时间: 2026-03-15 04:03
# =============================================================================

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'

# 配置文件路径
MIHOMO_CONFIG="$HOME/.config/mihomo/config.yaml"
BASHRC="$HOME/.bashrc"
STATUS_FILE="$HOME/.mihomo_status"
BACKUP_DIR="/media/lsk/future/proxy-config-backup/smart-backup"
LOG_FILE="$BACKUP_DIR/logs/mihomo-manager.log"

# 代理配置
PROXY_HOST="127.0.0.1"
PROXY_PORT="7890"
PROXY_VARS="HTTP_PROXY HTTPS_PROXY ALL_PROXY"

# 创建必要的目录
mkdir -p "$BACKUP_DIR"/{configs,logs,weekly,emergency}
mkdir -p "$(dirname "$LOG_FILE")"

# 日志函数
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
}

# 显示帮助信息
show_help() {
    echo -e "${BLUE}智能翻墙管理脚本${NC}"
    echo "用法: $0 [命令]"
    echo ""
    echo "命令:"
    echo "  on      开启翻墙"
    echo "  off     关闭翻墙"
    echo "  status  查看状态"
    echo "  backup  手动备份"
    echo "  restore 恢复配置"
    echo "  test    测试网络"
    echo "  help    显示帮助"
    echo ""
    echo "示例:"
    echo "  $0 on     # 开启翻墙"
    echo "  $0 off    # 关闭翻墙"
    echo "  $0 status # 查看当前状态"
}

# 检查权限
check_permissions() {
    if [ "$EUID" -eq 0 ]; then
        echo -e "${RED}错误: 请不要以root用户运行此脚本${NC}"
        exit 1
    fi
}

# 备份配置文件
backup_config() {
    local backup_name="backup_$(date +%Y%m%d_%H%M%S)"
    local backup_path="$BACKUP_DIR/configs/$backup_name"
    
    mkdir -p "$backup_path"
    
    # 备份关键文件
    cp "$BASHRC" "$backup_path/bashrc.bak" 2>/dev/null
    cp "$MIHOMO_CONFIG" "$backup_path/config.yaml.bak" 2>/dev/null
    
    # 记录当前状态
    echo "当前状态: $(get_status_simple)" > "$backup_path/status.txt"
    
    # 记录环境变量
    env | grep -i proxy > "$backup_path/env_vars.txt" 2>/dev/null
    
    log "配置已备份到: $backup_path"
    echo -e "${GREEN}✅ 配置已备份${NC}"
}

# 获取简单状态
get_status_simple() {
    if systemctl is-active --quiet mihomo; then
        service_status="running"
    else
        service_status="stopped"
    fi
    
    if [ -n "$HTTP_PROXY" ]; then
        env_status="proxy"
    else
        env_status="direct"
    fi
    
    echo "service:$service_status,env:$env_status"
}

# 测试网络连通性
test_network() {
    local test_result=0
    
    echo -e "${BLUE}测试网络连通性...${NC}"
    
    # 测试直连
    echo -n "直连测试 (百度): "
    if ping -c 1 -W 3 www.baidu.com >/dev/null 2>&1; then
        echo -e "${GREEN}✅ 正常${NC}"
    else
        echo -e "${RED}❌ 失败${NC}"
        test_result=1
    fi
    
    # 测试代理
    echo -n "代理测试 (Google): "
    if curl -s --connect-timeout 5 --proxy "http://$PROXY_HOST:$PROXY_PORT" \
        https://www.google.com >/dev/null 2>&1; then
        echo -e "${GREEN}✅ 正常${NC}"
    else
        echo -e "${RED}❌ 失败${NC}"
        test_result=1
    fi
    
    return $test_result
}

# 启用代理环境变量
enable_proxy_env() {
    # 检查bashrc中是否已有代理配置
    if ! grep -q "export HTTP_PROXY=" "$BASHRC"; then
        # 添加代理配置到bashrc末尾
        echo "" >> "$BASHRC"
        echo "# mihomo代理配置" >> "$BASHRC"
        echo "export HTTP_PROXY=http://$PROXY_HOST:$PROXY_PORT" >> "$BASHRC"
        echo "export HTTPS_PROXY=http://$PROXY_HOST:$PROXY_PORT" >> "$BASHRC"
        echo "export ALL_PROXY=http://$PROXY_HOST:$PROXY_PORT" >> "$BASHRC"
        log "代理配置已添加到 ~/.bashrc"
    else
        # 启用已有的代理配置（取消注释）
        sed -i 's/^#.*export HTTP_PROXY=/export HTTP_PROXY=/' "$BASHRC"
        sed -i 's/^#.*export HTTPS_PROXY=/export HTTPS_PROXY=/' "$BASHRC"
        sed -i 's/^#.*export ALL_PROXY=/export ALL_PROXY=/' "$BASHRC"
        log "代理配置已启用"
    fi
    
    # 设置当前会话的环境变量
    export HTTP_PROXY="http://$PROXY_HOST:$PROXY_PORT"
    export HTTPS_PROXY="http://$PROXY_HOST:$PROXY_PORT"
    export ALL_PROXY="http://$PROXY_HOST:$PROXY_PORT"
}

# 禁用代理环境变量
disable_proxy_env() {
    # 注释掉bashrc中的代理配置
    sed -i 's/^export HTTP_PROXY=/#export HTTP_PROXY=/' "$BASHRC"
    sed -i 's/^export HTTPS_PROXY=/#export HTTPS_PROXY=/' "$BASHRC"
    sed -i 's/^export ALL_PROXY=/#export ALL_PROXY=/' "$BASHRC"
    
    # 清除当前会话的环境变量
    unset HTTP_PROXY HTTPS_PROXY ALL_PROXY
    
    log "代理配置已禁用"
}

# 开启翻墙
enable_mihomo() {
    echo -e "${BLUE}开启翻墙...${NC}"
    
    # 备份当前配置
    backup_config
    
    # 启动mihomo服务
    echo "启动mihomo服务..."
    if sudo systemctl start mihomo; then
        echo -e "${GREEN}✅ mihomo服务已启动${NC}"
    else
        echo -e "${RED}❌ mihomo服务启动失败${NC}"
        return 1
    fi
    
    # 等待服务启动
    sleep 2
    
    # 启用代理环境变量
    enable_proxy_env
    
    # 保存状态
    echo "enabled" > "$STATUS_FILE"
    
    # 验证网络
    echo -e "${BLUE}验证网络连通性...${NC}"
    if test_network; then
        echo -e "${GREEN}🎉 翻墙已成功开启！${NC}"
        log "翻墙已开启"
    else
        echo -e "${YELLOW}⚠️ 翻墙已开启，但网络测试异常${NC}"
        log "翻墙开启但网络测试异常"
    fi
}

# 关闭翻墙
disable_mihomo() {
    echo -e "${BLUE}关闭翻墙...${NC}"
    
    # 备份当前配置
    backup_config
    
    # 停止mihomo服务
    echo "停止mihomo服务..."
    if sudo systemctl stop mihomo; then
        echo -e "${GREEN}✅ mihomo服务已停止${NC}"
    else
        echo -e "${RED}❌ mihomo服务停止失败${NC}"
        return 1
    fi
    
    # 禁用代理环境变量
    disable_proxy_env
    
    # 保存状态
    echo "disabled" > "$STATUS_FILE"
    
    # 验证网络
    echo -e "${BLUE}验证直连网络...${NC}"
    if ping -c 1 -W 3 www.baidu.com >/dev/null 2>&1; then
        echo -e "${GREEN}🎉 翻墙已关闭，直连网络正常！${NC}"
        log "翻墙已关闭，直连正常"
    else
        echo -e "${YELLOW}⚠️ 翻墙已关闭，但直连网络异常${NC}"
        log "翻墙关闭但直连异常"
    fi
}

# 显示状态
show_status() {
    echo -e "${BLUE}当前状态:${NC}"
    
    # 服务状态
    if systemctl is-active --quiet mihomo; then
        echo -e "mihomo服务: ${GREEN}运行中${NC}"
    else
        echo -e "mihomo服务: ${RED}已停止${NC}"
    fi
    
    # 环境变量
    if [ -n "$HTTP_PROXY" ]; then
        echo -e "环境变量: ${GREEN}代理模式${NC}"
        echo "  HTTP_PROXY=$HTTP_PROXY"
        echo "  HTTPS_PROXY=$HTTPS_PROXY"
    else
        echo -e "环境变量: ${YELLOW}直连模式${NC}"
    fi
    
    # 端口监听
    if ss -tlnp | grep -q ":7890"; then
        echo -e "7890端口: ${GREEN}监听中${NC}"
    else
        echo -e "7890端口: ${RED}未监听${NC}"
    fi
    
    # 状态文件
    if [ -f "$STATUS_FILE" ]; then
        saved_status=$(cat "$STATUS_FILE")
        echo -e "保存状态: $saved_status"
    else
        echo -e "保存状态: ${YELLOW}未设置${NC}"
    fi
}

# 恢复配置
restore_config() {
    echo -e "${BLUE}恢复配置...${NC}"
    
    # 列出可用的备份
    local backup_list=$(ls -1 "$BACKUP_DIR/configs" | grep "backup_" | sort -r)
    
    if [ -z "$backup_list" ]; then
        echo -e "${RED}❌ 没有找到可用的备份${NC}"
        return 1
    fi
    
    echo "可用的备份:"
    echo "$backup_list" | nl
    echo ""
    
    read -p "请选择要恢复的备份 (输入序号): " choice
    
    if [[ ! "$choice" =~ ^[0-9]+$ ]]; then
        echo -e "${RED}❌ 无效的选择${NC}"
        return 1
    fi
    
    local selected_backup=$(echo "$backup_list" | sed -n "${choice}p")
    
    if [ -z "$selected_backup" ]; then
        echo -e "${RED}❌ 备份不存在${NC}"
        return 1
    fi
    
    local backup_path="$BACKUP_DIR/configs/$selected_backup"
    
    # 恢复配置文件
    if [ -f "$backup_path/bashrc.bak" ]; then
        cp "$backup_path/bashrc.bak" "$BASHRC"
        echo -e "${GREEN}✅ ~/.bashrc 已恢复${NC}"
    fi
    
    if [ -f "$backup_path/config.yaml.bak" ]; then
        cp "$backup_path/config.yaml.bak" "$MIHOMO_CONFIG"
        echo -e "${GREEN}✅ mihomo配置已恢复${NC}"
    fi
    
    # 重新加载环境变量
    source "$BASHRC"
    
    log "配置已从备份恢复: $selected_backup"
    echo -e "${GREEN}🎉 配置恢复完成！${NC}"
}

# 开机检查函数
startup_check() {
    if [ -f "$STATUS_FILE" ]; then
        local saved_status=$(cat "$STATUS_FILE")
        if [ "$saved_status" = "enabled" ]; then
            # 确保mihomo运行
            sudo systemctl start mihomo 2>/dev/null
            # 设置环境变量
            enable_proxy_env
            log "开机检查: 已恢复翻墙状态"
        elif [ "$saved_status" = "disabled" ]; then
            # 确保mihomo停止
            sudo systemctl stop mihomo 2>/dev/null
            # 清除环境变量
            disable_proxy_env
            log "开机检查: 已保持关闭状态"
        fi
    else
        # 默认启用翻墙
        echo "enabled" > "$STATUS_FILE"
        enable_mihomo
        log "开机检查: 默认启用翻墙"
    fi
}

# 主函数
main() {
    # 检查权限
    check_permissions
    
    # 处理命令行参数
    case "${1:-help}" in
        "on")
            enable_mihomo
            ;;
        "off")
            disable_mihomo
            ;;
        "status")
            show_status
            ;;
        "backup")
            backup_config
            ;;
        "restore")
            restore_config
            ;;
        "test")
            test_network
            ;;
        "help"|*)
            show_help
            ;;
    esac
}

# 如果直接运行脚本
if [ "${BASH_SOURCE[0]}" = "${0}" ]; then
    main "$@"
fi
```

### 🔧 系统服务配置

#### mihomo.service
```ini
[Unit]
Description=Mihomo Proxy Service
After=network.target

[Service]
Type=simple
User=lsk
ExecStart=/usr/local/bin/mihomo -d /home/lsk/.config/mihomo

WantedBy=multi-user.target
```

#### 安装命令
```bash
# 创建服务文件
sudo tee /etc/systemd/system/mihomo.service > /dev/null << 'EOF'
```
